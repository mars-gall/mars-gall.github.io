# Ask the user for the first number num1 = input("Enter the first number: ") # Ask the user for the second number num2 = input("Enter the second number: ") # Convert the input (which is text) to numbers # We'll use "float" to allow for decimals num1_float = float(num1) num2_float = float(num2) # Add the two numbers together result = num1_float + num2_float # Print the result to the user print(f"The sum is: {result}")
